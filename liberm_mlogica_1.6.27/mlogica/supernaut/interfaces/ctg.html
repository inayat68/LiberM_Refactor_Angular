    CICS Transaction Gateway

    LIBER*TP provides native support for CICS Transaction Gateway (CTG) ECI through its own dedicated client libraries, available for Java and .NET.

    These libraries serve as a replacement for the standard IBM CTG client libraries and require minimal changes to the existing Java application code.

    LIBER*TP provides two types of CTG services (and corresponding client gateways):

    • REST Gateway Service based (HTTP/HTTPS):
      • this allows deployment behind a Layer 7 load balancer (e.g. Kubernetes Ingress)
      • supports only NO_EXTEND ECI requests
    • TCP Gateway Service (custom protocol over TCP sockets)
      • this allows deployment behind a Layer 4 load balancer (e.g. Kubernetes LoadBalancer)
      • supports both NO_EXTEND and EXTENDED ECI requests

    While the REST Gateway Service is active by default, TCP Gateway Services must be configured in the region as TCP services of type Ctg.

    The region must define one or more TCP services of type Ctg. Example configuration:

    YAML
    {
  "TcpServices": [
    {
      "serviceType": "Ctg",
      "name": "CTG",
      "address": "localhost",
      "port": "2325"
    },
    {
      "serviceType": "Ctg",
      "name": "CTGSSL",
      "address": "localhost",
      "port": "2326",
      "ssl": true,
      "keystore": "keystore.p12",
      "keystorePassword": "supersecret"
    }
  ]
}

    Java Client Libraries

    To incorporate the LIBER*TP CTG client libraries into your maven project, include the following Maven dependency:

    XML
    <dependency>
    <groupId>com.nib</groupId>
    <artifactId>nib-supernaut-ctg</artifactId>
    <version>${nib.version}</version>
</dependency>

    Java applications currently using IBM's CTG libraries can be easily refactored to use the LIBER*TP implementation with minimal effort.

    Two gateway types are provided to support different communication models:

    • REST Gateway: Connects to the LIBER*TP region over HTTP(S) using RESTful web services. This gateway supports only NO_EXTEND ECI requests.
    • TCP Gateway: Uses direct TCP socket communication with the LIBER*TP region and supports both NO_EXTEND and EXTENDED ECI requests.

    Both gateways communicate directly with the LIBER*TP region, eliminating the need for an intermediate gateway layer as required by IBM CICS™. SSL is supported by both gateway types.

    REST Gateway

    The RestSupernautGateway class provides a CTG-compatible interface over HTTP/HTTPS, leveraging REST principles. This enables deployment scenarios such as exposing the gateway behind a Layer 7 load balancer (e.g., a Kubernetes Ingress).

    Due to the stateless nature of REST, this gateway supports only ECI_NO_EXTEND requests. No specific configuration is required on the region side—LIBER*TP exposes the transactionGateway endpoint by default. This endpoint can be served behind the same load balancer used for other LIBER*TP REST APIs.

    Note: Only synchronous (ECI_SYNC) calls are supported.

    Example:

    Java
    String rest() throws IOException {
    RestSupernautGateway gw = new RestSupernautGateway("http://liber-tp.mlogica.net:8080/transactionGateway");

    var inputCommarea = "this is an example".getBytes();
    var req = new ECIRequest(
            ECIRequest.ECI_SYNC,
            username,
            password,
            "CTGCAPGM",
            "CSOL",
            inputCommarea,
            ECIRequest.ECI_NO_EXTEND,
            ECIRequest.ECI_LUW_NEW
    );
    rc = gw.flow(req);
    assertEquals(0, rc);
    assertEquals(0, req.getGatewayRc());
    assertEquals(0, req.getCicsRc());
    return new String(inputCommarea);
}

    TCP Gateway

    The TcpSupernautGateway class offers a CTG-compatible interface over a custom, stateful TCP protocol. It can be deployed behind a Layer 4 load balancer (e.g., a Kubernetes LoadBalancer).

    This gateway supports a broader set of ECI request types, including:

    • NO_EXTEND
    • EXTENDED
    • BACKOUT
    • COMMIT

    Note: Only synchronous (ECI_SYNC) communication is supported.

    Example:

    Java
    String tcpCics() throws IOException {
    Properties sslprops = new Properties();
    sslprops.setProperty(TcpSupernautGateway.SSL_PROP_KEYSTORE_PATH, "keystore.p12");
    sslprops.setProperty(TcpSupernautGateway.SSL_PROP_KEYSTORE_PASSWORD, "supersecret");

    ISupernautGateway gw = new TcpSupernautGateway("localhost", 2326, sslprops);

    var inputCommarea = "this is an example".getBytes();
    var req = new ECIRequest(
            ECIRequest.ECI_SYNC,
            username,
            password,
            "CTGCAPGM",
            "CSOL",
            inputCommarea,
            ECIRequest.ECI_NO_EXTEND,
            ECIRequest.ECI_LUW_NEW
    );
    rc = gw.flow(req);
    assertEquals(0, rc);
    assertEquals(0, req.getGatewayRc());
    assertEquals(0, req.getCicsRc());
    return new String(inputCommarea);
}

    Channels and Containers

    Both REST and TCP gateways support the use of channels and containers for structured data exchange. The following example demonstrates how to use these features with the TCP gateway:

    Java
    public final static String C1_VALUE = "1234567890";
public final static String C2_VALUE = "ABCDEFGHILMNO";

public void ctgTcpSslEciChannel() throws IOException {
    Properties sslprops = new Properties();
    sslprops.setProperty(TcpSupernautGateway.SSL_PROP_KEYSTORE_PATH, "keystore.p12");
    sslprops.setProperty(TcpSupernautGateway.SSL_PROP_KEYSTORE_PASSWORD, "lafavaelarava");
    ISupernautGateway gw = new TcpSupernautGateway("localhost", 2326, sslprops);

    var channel = new Channel(CTGCCPGM.CHANNEL_NAME);

    channel.createContainer(CTGCCPGM.CONTAINER_01_NAME, CTGCCPGM.C1_VALUE);
    channel.createContainer(CTGCCPGM.CONTAINER_02_NAME, CTGCCPGM.C2_VALUE);

    var req = new ECIRequest(
            ECIRequest.ECI_SYNC,
            "REGTEST",
            "regtest",
            "CTGCCPGM",
            "CSOL",
            channel,
            ECIRequest.ECI_NO_EXTEND,
            ECIRequest.ECI_LUW_NEW
    );
    int rc = gw.flow(req);
    assertEquals(0, rc);
    assertEquals(0, req.getGatewayRc());
    assertEquals(0, req.getCicsRc());

    System.out.println("CICS CHANNEL: " + req.getChannel().getName());
    for (var c : req.getChannel().getContainers()) {
        System.out.println("CICS CONTAINER NAME: " + c.getName());
        System.out.println("CICS CONTAINER DATA: " + new String(c.getData()));
    }
}

    .NET Client Libraries

    Text Only
    dotnet add <your-project>.csproj package Nib.Supernaut.CTG

    .NET applications currently using IBM's CTG libraries can be easily refactored to use the LIBER*TP implementation with minimal effort.

    • TCP Gateway: Uses direct TCP socket communication with the LIBER*TP region and supports both NO_EXTEND and EXTENDED ECI requests.

    REST Gateway

    The RestGatewayConnection class provides a CTG-compatible interface over HTTP/HTTPS, leveraging REST principles. This enables deployment scenarios such as exposing the gateway behind a Layer 7 load balancer (e.g., a Kubernetes Ingress).

    Due to the stateless nature of REST, this gateway supports only ECI_NO_EXTEND requests. No specific configuration is required on the region side—LIBER*TP exposes the transactionGateway endpoint by default. This endpoint can be served behind the same load balancer used for other LIBER*TP REST APIs.

    Note: Only synchronous (ECI_SYNC) calls are supported.

    Example:

    C#
    public void CtgRestEciNoExtendCommarea()
{
    GatewayConnection gw = new RestGatewayConnection
    {
        URL = "http://127.0.0.1:8080/transactionGateway"
    };

    var ca = Encoding.UTF8.GetBytes(CTGCAPGM.INPUT_CA);

    var req = new EciRequest();
    req.SyncMode = EciRequest.ECI_SYNC;
    req.UserId = "REGTEST";
    req.Password = "regtest";
    req.Program = "CTGCAPGM";
    req.TransId = "CSOL";
    req.ExtendMode = EciExtendMode.EciNoExtend;
    req.LuwToken = EciRequest.ECI_LUW_NEW;
    req.SetCommareaData(ca);

    int rc = 0;
    try
    {
        rc = gw.Flow(req);
    }
    catch (IOException e)
    {
        Assert.Fail(e.ToString());
    }

    Assert.That(rc, Is.EqualTo(0));
    Assert.That(req.GatewayReturnCode, Is.EqualTo(GatewayReturnCode.CtgNoError));
    Assert.That(req.EciReturnCode, Is.EqualTo(EciReturnCode.EciNoError));
    Assert.That(Encoding.UTF8.GetString(req.GetCommareaData()), Is.EqualTo(CTGCAPGM.RETURNED_CA));
}

    Channels and Containers

    Both REST and TCP gateways support the use of channels and containers for structured data exchange. The following example demonstrates how to use these features with the TCP gateway:

    C#
    public void CtgTcpSslEciChannel()
{
    Dictionary<string, string> sslprops = new();
    sslprops[TcpGatewayConnection.SSL_PROP_KEYSTORE_PATH] = "keystore.p12";
    sslprops[TcpGatewayConnection.SSL_PROP_KEYSTORE_PASSWORD] = "lafavaelarava";
    GatewayConnection gw = new TcpGatewayConnection("localhost", 2326, sslprops);
    var channel = new Nib.Supernaut.Ctg.Channel(CTGCCPGM.CHANNEL_NAME);
    channel.CreateContainer(CTGCCPGM.CONTAINER_01_NAME, CTGCCPGM.C1_VALUE);
    channel.CreateContainer(CTGCCPGM.CONTAINER_02_NAME, CTGCCPGM.C2_VALUE);

    var req = new EciRequest();
    req.SyncMode = EciRequest.ECI_SYNC;
    req.UserId = "REGTEST";
    req.Password = "regtest";
    req.Program = "CTGCCPGM";
    req.TransId = "CSOL";
    req.ExtendMode = EciExtendMode.EciNoExtend;
    req.LuwToken = EciRequest.ECI_LUW_NEW;
    req.Channel = channel;

    int rc = 0;
    try
    {
        rc = gw.Flow(req);
    }
    catch (IOException e)
    {
        Assert.Fail(e.ToString());
    }

    Assert.That(rc, Is.EqualTo(0));
    Assert.That(req.GatewayReturnCode, Is.EqualTo(GatewayReturnCode.CtgNoError));
    Assert.That(req.EciReturnCode, Is.EqualTo(EciReturnCode.EciNoError));

    Console.WriteLine("CICS CHANNEL : " + req.Channel.Name);
    Assert.That(req.Channel.Name, Is.EqualTo(CTGCCPGM.CHANNEL_NAME));
    Assert.That(req.Channel.Containers.Count, Is.EqualTo(3));
    foreach (var c in req.Channel.Containers)
    {
        if (c.Name.Equals(CTGCCPGM.CONTAINER_01_NAME))
        {
            Assert.That(Encoding.UTF8.GetString(c.Data), Is.EqualTo(CTGCCPGM.C1_VALUE));
        }
        else if (c.Name.Equals(CTGCCPGM.CONTAINER_03_NAME))
        {
            Assert.That(Encoding.UTF8.GetString(c.Data), Is.EqualTo(CTGCCPGM.C3_VALUE));
        }
        else if (c.Name.Equals(CTGCCPGM.CONTAINER_02_NAME))
        {
            //assertEquals("Server response #2", new String(c.getData()));
        }
        else
        {
            Assert.Fail("Unexpected container: " + c.Name);
        }
    }
}
